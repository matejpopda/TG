\begin{lemma}
    $G=(V,E)$ k-souvislý, $X\subseteq V , |X|\geq k$, přidáme vrchol $v$ který je spojený s vrcholy v $X$, tj stupně alespoň $k$.

    Pak je $G+v$ $k$-souvislý.
\end{lemma}
